      currentImages = images.length > 0 ? images : [''];
      currentImageIndex = 0;
      isImageZoomed = false;

      const colorsData = product.colors ? JSON.parse(product.colors) : [];

      selectedColorInDetail = colorsData.length > 0 ? colorsData[0] : null;

      // Render thumbnails
      renderThumbnails();

        <div class="px-4 py-2 rounded-xl inline-block" style="background: ${product.stock > 0 ? '#dbeafe' : '#fee2e2'};">

    function renderMainImageViewer() {
      const mainViewer = document.getElementById('main-image-viewer');
      const currentUrl = currentImages[currentImageIndex];
      const showArrows = currentImages.length > 1 && currentImages[0] !== '';

      if (currentUrl) {
        mainViewer.innerHTML = `
          ${showArrows ? `
            <button onclick="previousImage()" class="nav-arrow absolute left-2 top-1/2 -translate-y-1/2 z-10 bg-white/90 rounded-full p-3 shadow-lg" style="color: #000000;">
              <svg class="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M15 19l-7-7 7-7"/>
              </svg>
            </button>
            <button onclick="nextImage()" class="nav-arrow absolute right-2 top-1/2 -translate-y-1/2 z-10 bg-white/90 rounded-full p-3 shadow-lg" style="color: #000000;">
              <svg class="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 5l7 7-7 7"/>
              </svg>
            </button>
          ` : ''}
          <div class="image-viewer flex items-center justify-center min-h-[400px]" onclick="toggleZoom(event)" onmousemove="handleZoomPan(event)">
            <img src="${currentUrl}" alt="${selectedProductForDetail.name}" class="max-w-full max-h-[600px] w-auto h-auto object-contain" loading="lazy" onerror="this.style.display='none'; this.nextElementSibling.style.display='flex';">
            <div class="hidden w-full h-full items-center justify-center text-8xl">🛍️</div>
          </div>
          ${showArrows ? `
            <div class="absolute bottom-4 left-1/2 -translate-x-1/2 bg-black/60 text-white px-3 py-1 rounded-full text-sm">
              ${currentImageIndex + 1} / ${currentImages.length}
            </div>
          ` : ''}
        `;
      } else {
        mainViewer.innerHTML = `
          <div class="flex items-center justify-center min-h-[400px] text-8xl">🛍️</div>
        `;
      }
    }

    function renderThumbnails() {
      const container = document.getElementById('thumbnails-container');
      
      if (currentImages.length <= 1 || currentImages[0] === '') {
        container.innerHTML = '';
        return;
      }

      container.innerHTML = currentImages.map((url, index) => `
        <div onclick="selectImage(${index})" class="thumbnail-img w-20 h-20 rounded-lg border-2 overflow-hidden flex-shrink-0 ${index === currentImageIndex ? 'active' : ''}" style="border-color: ${index === currentImageIndex ? config.primary_color : '#bfdbfe'};">
          <img src="${url}" alt="Imagen ${index + 1}" class="w-full h-full object-cover" loading="lazy" onerror="this.style.display='none'; this.parentElement.innerHTML='<div class=\'w-full h-full flex items-center justify-center text-2xl\'>🛍️</div>';">
        </div>
      `).join('');
    }

    function selectImage(index) {
      currentImageIndex = index;
      isImageZoomed = false;
      renderMainImageViewer();
      renderThumbnails();
    }

    function previousImage() {
      currentImageIndex = (currentImageIndex - 1 + currentImages.length) % currentImages.length;
      isImageZoomed = false;
      renderMainImageViewer();
      renderThumbnails();
    }

    function nextImage() {
      currentImageIndex = (currentImageIndex + 1) % currentImages.length;
      isImageZoomed = false;
      renderMainImageViewer();
      renderThumbnails();
    }

    function toggleZoom(event) {
      const viewer = event.currentTarget;
      const img = viewer.querySelector('img');
      
      if (!img || img.style.display === 'none') return;

      isImageZoomed = !isImageZoomed;
      
      if (isImageZoomed) {
        viewer.classList.add('zoomed');
      } else {
        viewer.classList.remove('zoomed');
        img.style.transformOrigin = 'center center';
      }
    }

    function handleZoomPan(event) {
      const viewer = event.currentTarget;
      const img = viewer.querySelector('img');
      
      if (!isImageZoomed || !img || img.style.display === 'none') return;

      const rect = viewer.getBoundingClientRect();
      const x = ((event.clientX - rect.left) / rect.width) * 100;
      const y = ((event.clientY - rect.top) / rect.height) * 100;
      
      img.style.transformOrigin = `${x}% ${y}%`;
    }

    function selectSizeInDetail(size) {
      selectedSizeInDetail = size;
      document.querySelectorAll('.size-btn').forEach(btn => {
        if (btn.dataset.size === size) {
          btn.style.borderColor = config.primary_color;
          btn.style.background = config.primary_color;
          btn.style.color = 'white';
        } else {
          btn.style.borderColor = '#bfdbfe';
          btn.style.background = 'white';
          btn.style.color = '#000000';
        }
      });
    }

    function selectColorInDetail(colorObj) {
      selectedColorInDetail = colorObj;
      document.querySelectorAll('.color-btn').forEach(btn => {
        if (btn.dataset.color === colorObj.hex) {
          btn.style.borderColor = config.primary_color;
        } else {
          btn.style.borderColor = '#bfdbfe';
        }
      });
    }

    function closeProductDetail() {
      document.getElementById('product-detail-modal').classList.add('hidden');
      document.body.style.overflow = '';
      selectedProductForDetail = null;
      selectedSizeInDetail = null;
      selectedColorInDetail = null;
      currentImageIndex = 0;
      currentImages = [];
      isImageZoomed = false;
    }

    function addToCartFromDetail() {
      if (!selectedProductForDetail) return;
      if (selectedProductForDetail.stock <= 0) {
        showToast('⚠️ Producto agotado');
        return;
      }

      const cartItem = {
        ...selectedProductForDetail,
        cartId: Date.now(),
        selectedSize: selectedSizeInDetail,
        selectedColor: selectedColorInDetail,
        quantity: 1
      };

      cart.push(cartItem);
      updateCartUI();
      showToast(`${selectedProductForDetail.name} agregado al carrito`);
      closeProductDetail();
    }

    // Filter by category
    function filterCategory(category) {
      currentCategory = category;
      renderProducts();
      renderCategories();
    }

    // Remove from cart
    function removeFromCart(cartId) {
      cart = cart.filter(item => item.cartId !== cartId);
      updateCartUI();
    }

    // Update quantity
    function updateQuantity(cartId, delta) {
      const item = cart.find(i => i.cartId === cartId);
      if (item) {
        item.quantity = Math.max(1, item.quantity + delta);
        updateCartUI();
      }
    }

    // Update cart UI
    function updateCartUI() {
      const cartCount = document.getElementById('cart-count');
      const cartItems = document.getElementById('cart-items');
      const cartTotal = document.getElementById('cart-total');
      const checkoutBtn = document.getElementById('checkout-btn');

      const totalItems = cart.reduce((sum, item) => sum + item.quantity, 0);
      cartCount.textContent = totalItems;

      if (cart.length === 0) {
        cartItems.innerHTML = `
          <div class="flex flex-col items-center justify-center h-full text-center">
            <span class="text-6xl mb-4">      </span>
            <p class="font-medium" style="color: #000000;">Tu carrito está vacío</p>
            <p class="text-sm mt-2" style="color: #000000;">¡Agrega productos para comenzar!</p>
          </div>
        `;
        checkoutBtn.disabled = true;
      } else {
        cartItems.innerHTML = cart.map(item => {
          const images = item.images ? item.images.split('\n').filter(url => url.trim()) : [];
          const firstImage = images[0] || '';
          
          return `
          <div class="flex gap-4 p-4 rounded-xl mb-4" style="background: #f0f9ff;">
            <div class="w-16 h-16 rounded-xl flex items-center justify-center overflow-hidden" style="background: white;">
              ${firstImage ? `
                <img src="${firstImage}" alt="${item.name}" class="w-full h-full object-cover" loading="lazy" onerror="this.style.display='none'; this.nextElementSibling.style.display='block';">
                <span class="hidden text-3xl">🛍️</span>
              ` : `
                <span class="text-3xl">🛍️</span>
              `}
            </div>
            <div class="flex-1">
              <h4 class="font-medium" style="color: #000000;">${item.name}</h4>
              <div class="text-xs mt-1" style="color: #000000;">
                ${item.selectedSize ? `Talla: ${item.selectedSize}` : ''}
                ${item.selectedSize && item.selectedColor ? ' | ' : ''}
                ${item.selectedColor ? `Color: ${item.selectedColor.name}` : ''}
              </div>
              <div class="flex items-center justify-between mt-2">
                <div class="flex items-center gap-2">
                  <button onclick="updateQuantity(${item.cartId}, -1)" class="w-6 h-6 rounded-full flex items-center justify-center text-white text-sm" style="background: ${config.primary_color};">-</button>
                  <span class="font-medium">${item.quantity}</span>
                  <button onclick="updateQuantity(${item.cartId}, 1)" class="w-6 h-6 rounded-full flex items-center justify-center text-white text-sm" style="background: ${config.primary_color};">+</button>
                </div>
                <span class="font-semibold" style="color: #000000;">${formatPrice(item.price * item.quantity)}</span>
              </div>
            </div>
            <button onclick="removeFromCart(${item.cartId})" class="text-red-400 hover:text-red-600 transition-colors">
              <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"/>
              </svg>
            </button>
          </div>
        `;
        }).join('');
        checkoutBtn.disabled = false;
      }

      const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
      cartTotal.textContent = formatPrice(total);
    }

    // Show toast notification
    function showToast(message) {
      const container = document.getElementById('toast-container');
      const toast = document.createElement('div');
      toast.className = 'toast px-6 py-3 rounded-full shadow-lg text-white font-medium mb-2';
      toast.style.background = config.primary_color;
      toast.textContent = message;
      container.appendChild(toast);
      setTimeout(() => toast.remove(), 3000);
    }

    // Submit order to WhatsApp
    function submitOrder(e) {
      e.preventDefault();
      
      let message = `Hola, quiero hacer un pedido.\n\n`;
      
      const template = config.shipping_template || defaultConfig.shipping_template || 'home';
      
      if (template === 'home') {
        const name = document.getElementById('customer-name').value;
        const phone = document.getElementById('customer-phone').value;
        const province = document.getElementById('customer-province').value;
        const department = document.getElementById('customer-department').value;
        const district = document.getElementById('customer-district').value;
        const address = document.getElementById('customer-address').value;
        const notes = document.getElementById('customer-notes').value;

        message += `👤 *Datos del Cliente (Envío a Domicilio):*\n`;
        message += `• Nombre completo: ${name}\n`;
        message += `• Teléfono: ${phone}\n`;
        message += `• Departamento: ${department}\n`;
        message += `• Provincia: ${province}\n`;
        message += `• Distrito/Ciudad: ${district}\n`;
        message += `• Dirección Completa: ${address}\n`;
        if (notes) message += `• Notas: ${notes}\n`;
      } else {
        const name = document.getElementById('agency-customer-name').value;
        const dni = document.getElementById('agency-customer-dni').value;
        const city = document.getElementById('agency-customer-city').value;
        const agency = document.getElementById('agency-customer-agency').value;
        const phone = document.getElementById('agency-customer-phone').value;
        const notes = document.getElementById('agency-customer-notes').value;
        const agencyName = config.agency_name || defaultConfig.agency_name || 'Agencia Shalom';

        message += `👤 *Datos del Cliente (Envío por Agencia):*\n`;
        message += `• Nombre completo: ${name}\n`;
        message += `    DNI: ${dni}\n`;
        message += `• Ciudad: ${city}\n`;
        message += `• ${agencyName}: ${agency}\n`;
        message += `• Celular: ${phone}\n`;
        if (notes) message += `• Notas: ${notes}\n`;
      }
      
      message += `\n📦 *Productos:*\n`;
      message += `─────────────────\n`;
      
      cart.forEach((item, index) => {
        message += `${index + 1}. *${item.name}* (${item.code})\n`;
        
        // Add product image URL if available
        const images = item.images ? item.images.split('\n').filter(url => url.trim()) : [];
        if (images.length > 0) {
          message += `   🔗 URL: ${images[0]}\n`;
        }
        
        if (item.selectedSize) message += `   Talla: ${item.selectedSize}\n`;
        if (item.selectedColor) message += `   Color: ${item.selectedColor.name}\n`;

      });

      }
